Drop table flight;

CREATE TABLE `flight_reservation`.`user` (
  `pid` INT NOT NULL AUTO_INCREMENT,
  `fname` VARCHAR(45) NOT NULL,
  `lname` VARCHAR(45) NOT NULL,
  `gender` VARCHAR(45) NOT NULL,
  `age` INT NOT NULL,
  `contactNo` VARCHAR(45) NOT NULL,
  `address` VARCHAR(45) NOT NULL,
  `emailid` VARCHAR(45) NOT NULL,
  `password` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`pid`),
  UNIQUE INDEX `pid_UNIQUE` (`pid` ASC) VISIBLE);



CREATE TABLE `airlinereservationsystem`.`flight` (
  `flightId` INT NOT NULL AUTO_INCREMENT,
  `flightName` VARCHAR(45) NOT NULL,
  `flightNumber` VARCHAR(45) NOT NULL,
`source` VARCHAR(45) NOT NULL,
  `destination` VARCHAR(45) NOT NULL,
  `noOfStops` VARCHAR(45) NOT NULL,
`date` VARCHAR(45)NOT NULL,
`time` VARCHAR(45)NOT NULL,
`price` DOUBLE NOT NULL,
  PRIMARY KEY (`flightId`),
  UNIQUE INDEX `flightId_UNIQUE` (`flightId` ASC) VISIBLE);

CREATE TABLE `airlinereservationsystem`.`booking` (
  `bookingid` INT NOT NULL AUTO_INCREMENT,
  `pid` INT NOT NULL,
  `flight` INT NOT NULL,
  `date` VARCHAR(45) NOT NULL,
  `time` VARCHAR(45) NOT NULL,
  `b_price` DOUBLE NOT NULL,
  PRIMARY KEY (`bookingid`),
  UNIQUE INDEX `bookingid_UNIQUE` (`bookingid` ASC) VISIBLE,
  INDEX `fk_pid_idx` (`pid` ASC) VISIBLE,
  INDEX `fk_flightid_idx` (`flight` ASC) VISIBLE,
  CONSTRAINT `fk_pid`
    FOREIGN KEY (`pid`)
    REFERENCES `airlinereservationsystem`.`user` (`pid`)
    ON DELETE NO ACTION
    ON UPDATE CASCADE,
  CONSTRAINT `fk_flightid`
    FOREIGN KEY (`flight`)
    REFERENCES `airlinereservationsystem`.`flight` (`flightId`)
    ON DELETE NO ACTION
    ON UPDATE CASCADE);
