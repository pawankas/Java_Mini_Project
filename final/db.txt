

CREATE TABLE `airlinereservationsystem`.`user` (
`pid` INT NOT NULL AUTO_INCREMENT,
`fname` VARCHAR(45) NOT NULL,
`lname` VARCHAR(45) NOT NULL,
`gender` VARCHAR(45) NOT NULL,
`age` INT NOT NULL,
`contactNo` VARCHAR(45) NOT NULL,
`address` VARCHAR(45) NOT NULL,
`emailid` VARCHAR(45) NOT NULL,
`password` VARCHAR(45) NOT NULL,
`role` VARCHAR(45) NOT NULL,
PRIMARY KEY (`pid`));


create table `airlinereservationsystem`.`flight`(
flightId int NOT NULL AUTO_INCREMENT,
flightName varchar(50),
flightNumber varchar(45) NOT NULL,
destination varchar(25),
noOfStops varchar(45),
Primary KEY(flightId)
);


create table `airlinereservationsystem`.`flightDetails`(
flightDetailsId int NOT NULL AUTO_INCREMENT,
flightId int NOT NULL,
arrival Datetime,
departure Datetime,
destination varchar(25),
fromSource varchar(25),
Economy int,
Business int,
Premimum int,
Capacity int,
Primary KEY(flightDetailsId),
FOREIGN KEY (flightId) REFERENCES flight(flightId)
);
